-- 1. EXTENSIONS
create extension if not exists "uuid-ossp";

-- 2. PROFILES TABLE (Links to Supabase Auth)
create table profiles (
  id uuid references auth.users on delete cascade primary key,
  role text check (role in ('admin', 'dosen')) default 'dosen',
  full_name text,
  nidn text unique,
  updated_at timestamp with time zone default timezone('utc'::text, now())
);

-- 3. DOSEN TABLE
create table dosen (
  id text primary key,
  nama text not null,
  nidn text unique not null,
  foto_url text,
  jabatan text,
  pangkat text,
  email text,
  telepon text,
  bidang_keahlian text[],
  program_studi text default 'D4 Teknik Listrik',
  pendidikan_terakhir text,
  created_at timestamp with time zone default timezone('utc'::text, now())
);

-- 4. KARYA TABLE
create table karya (
  id uuid default uuid_generate_v4() primary key,
  dosen_id text references dosen(id) on delete cascade,
  jenis text not null check (jenis in ('publikasi', 'penelitian', 'pengabdian', 'bukuAjar', 'hki', 'sertifikasi')),
  judul text not null,
  tahun integer not null,
  deskripsi text,
  metadata jsonb,
  created_at timestamp with time zone default timezone('utc'::text, now())
);

-- 5. GALERI TABLE (Removed 'warna')
create table galeri (
  id uuid default uuid_generate_v4() primary key,
  judul text not null,
  deskripsi text,
  tanggal date not null,
  kategori text check (kategori in ('fasilitas', 'tridharma')),
  foto_urls text[],
  created_at timestamp with time zone default timezone('utc'::text, now())
);

-- 6. KURIKULUM TABLES
create table kurikulum_aktif (
  id integer primary key default 1,
  nama text not null,
  deskripsi text,
  berlaku_sejak text,
  file_url text,
  constraint single_row check (id = 1)
);

create table mata_kuliah (
  kode text primary key,
  nama text not null,
  sks integer not null,
  semester integer not null,
  jenis text check (jenis in ('Teori', 'Praktik', 'Teori & Praktik'))
);

create table cpl (
  kode text primary key,
  deskripsi text not null
);

-- 7. ROW LEVEL SECURITY (RLS) POLICIES

alter table profiles enable row level security;
alter table dosen enable row level security;
alter table karya enable row level security;
alter table galeri enable row level security;
alter table kurikulum_aktif enable row level security;
alter table mata_kuliah enable row level security;
alter table cpl enable row level security;

-- Public Read Access
create policy "Allow public read access on dosen" on dosen for select using (true);
create policy "Allow public read access on karya" on karya for select using (true);
create policy "Allow public read access on galeri" on galeri for select using (true);
create policy "Allow public read access on kurikulum" on kurikulum_aktif for select using (true);
create policy "Allow public read access on mata_kuliah" on mata_kuliah for select using (true);
create policy "Allow public read access on cpl" on cpl for select using (true);

-- Admin Access
create policy "Admins can do everything on profiles" on profiles for all using (
  (select role from profiles where id = auth.uid()) = 'admin'
);
create policy "Admins can do everything on dosen" on dosen for all using (
  (select role from profiles where id = auth.uid()) = 'admin'
);
create policy "Admins can do everything on galeri" on galeri for all using (
  (select role from profiles where id = auth.uid()) = 'admin'
);
create policy "Admins can do everything on karya" on karya for all using (
  (select role from profiles where id = auth.uid()) = 'admin'
);

-- Dosen Access
create policy "Dosen can update their own data" on dosen for update using (
  auth.uid() in (select id from profiles where role = 'dosen' and nidn = dosen.nidn)
);
create policy "Dosen can manage their own karya" on karya for all using (
  auth.uid() in (select id from profiles where role = 'dosen' and nidn = (select nidn from dosen where id = karya.dosen_id))
);
